+    def _execute_sequential(self, function, data):
+        """Execute functions sequentially in the main thread.
+
+        This method is used when num_threads=1 to ensure that all execution
+        happens in the main thread, which is required by some libraries like
+        AppWorld that use signals and other main-thread-only functionality.
+        """

+        pbar = tqdm.tqdm(
+            total=len(data),
+            dynamic_ncols=True,
+            disable=self.disable_progress_bar,
+            file=sys.stdout,
+        )
+
+        for item in data:
+            if self.cancel_jobs.is_set():
+                break
+            try:
+                result = function(item)
+                results.append(result)
+                pbar.update()
+            except Exception as e:
+                with self.error_lock:
+                    self.error_count += 1
+                    if self.error_count >= self.max_errors:
+                        self.cancel_jobs.set()

+                results.append(e)
+                pbar.update()
+
+        pbar.close()
+
+        if self.cancel_jobs.is_set():
+            logger.warning("Execution cancelled due to errors or interruption.")
+            raise Exception("Execution cancelled due to errors or interruption.")
+
+        return results
