+_STATE_CLASS_REGISTRY: dict[str, Type["BaseState"]] = {}

+def _register_state_class(cls: Type["BaseState"]) -> None:
+    """Register a State subclass for later lookup during pickling.
+
+    If the class was defined inside a local function its ``__qualname__`` will
+    contain ``<locals>`` which makes it unpickleable by default.  By forcing the
+    ``__module__`` attribute to a known importable module (``reflex.state``) and
+    storing the class in a registry we can reconstruct it safely.
+    """
+    # Build a fully‑qualified identifier.
+    fq_name = f"{cls.__module__}.{cls.__qualname__}"
+    _STATE_CLASS_REGISTRY[fq_name] = cls
+
+    # If the class originates from a local scope, adjust its module so that
+    # ``pickle`` can locate it via ``import reflex.state``.
+    if "<locals>" in cls.__qualname__:
+        cls.__module__ = "reflex.state"
