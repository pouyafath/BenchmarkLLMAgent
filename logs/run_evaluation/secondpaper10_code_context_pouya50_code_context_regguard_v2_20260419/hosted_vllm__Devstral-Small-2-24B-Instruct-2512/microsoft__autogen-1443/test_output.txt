+    match = src_pattern.search(img_tag)

+    # Then try to match <img src=...> without quotes (unquoted attribute)
+    src_unquoted_pattern = re.compile(r'<img\s+src=([^\s>]+)', re.IGNORECASE)

+
+    # Then try to match <img ...> where the rest is the path
