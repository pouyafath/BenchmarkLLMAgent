+    - If GATEWAY_PROTOCOL is set to "http", use HTTP
+    - If GATEWAY_PROTOCOL is "auto" (default), try to detect from environment
+    - Falls back to HTTP for backward compatibility

+    Returns:

+    
+    if protocol == "https":
+        return "https"
+    elif protocol == "http":
+        return "http"

+        # Try to detect from external_url if set
+        if settings.external_url:
+            from urllib.parse import urlparse
+            parsed = urlparse(settings.external_url)
+            return parsed.scheme or "http"
+        
+        # Default to http for backward compatibility
+        return "http"
+    else:
+        # Invalid value, default to http
+        logger.warning(f"Invalid GATEWAY_PROTOCOL value: {protocol}. Using http as default.")
+        return "http"
