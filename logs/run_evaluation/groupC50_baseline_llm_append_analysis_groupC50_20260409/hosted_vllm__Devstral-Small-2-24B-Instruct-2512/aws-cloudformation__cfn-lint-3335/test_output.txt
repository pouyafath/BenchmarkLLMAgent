+def _split_dynamic_reference(dynamic_ref):
+    '''Split a dynamic reference string on ':' but ignore colons inside ${...} expressions.'''
+    parts = []
+    current_part = ''
+    i = 0
+    while i < len(dynamic_ref):
+        if dynamic_ref[i] == '$' and i + 1 < len(dynamic_ref) and dynamic_ref[i+1] == '{':
+            # Found start of ${...} expression
+            current_part += dynamic_ref[i]
+            i += 1
+            # Skip until we find the closing }
+            while i < len(dynamic_ref) and dynamic_ref[i] != '}':
+                current_part += dynamic_ref[i]
+                i += 1
+            if i < len(dynamic_ref):
+                current_part += dynamic_ref[i]  # Add the closing }
+                i += 1
+        elif dynamic_ref[i] == ':':
+            # Found a colon - add current part and start new one
+            parts.append(current_part)
+            current_part = ''
+            i += 1
+        else:
+            current_part += dynamic_ref[i]
+            i += 1
+    # Add the last part if it's not empty, or if we ended with a colon
+    if current_part or (i > 0 and dynamic_ref[i-1] == ':'):
+        parts.append(current_part)
+    return parts
