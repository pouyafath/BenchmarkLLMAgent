+# Protocol detection helper
+def get_protocol_from_request(request: Request) -> str:
+    """Detect protocol from request headers and context.
+
+    Checks for HTTPS indicators in the following order:
+    1. X-Forwarded-Proto header (common with reverse proxies)
+    2. request.url.scheme (direct HTTPS)
+    3. X-Forwarded-SSL header
+    4. Default to http
+
+    Args:
+        request: FastAPI Request object
+
+    Returns:
+        str: "http" or "https"
+    """
+    # Check X-Forwarded-Proto header (most reliable for reverse proxies)
+    forwarded_proto = request.headers.get("X-Forwarded-Proto")
+    if forwarded_proto:
+        return forwarded_proto.lower()
+
+    # Check if request is secure (direct HTTPS)
+    if hasattr(request, 'url') and hasattr(request.url, 'scheme'):
+        if request.url.scheme == "https":
+            return "https"
+
+    # Check for other proxy headers
+    if request.headers.get("X-Forwarded-SSL") == "on":
+        return "https"
+
+    # Default to http
+    return "http"
+
+def update_url_protocol(url: str, request: Request = None) -> str:
+    """Update URL protocol based on request context or configuration.
+
+    If a request is provided, detects the protocol from the request.
+    Otherwise, uses the gateway_protocol setting from configuration.
+
+    Args:
+        url: The URL to update
+        request: Optional FastAPI Request object for protocol detection
+
+    Returns:
+        str: URL with updated protocol
+    """
+    if request:
+        # Detect protocol from request
+        protocol = get_protocol_from_request(request)
+    else:
+        # Use configuration
+        protocol = settings.gateway_protocol
+        if protocol == "auto":
+            protocol = "http"  # Default to http if auto-detection not possible
+
+    # Only update if URL has a protocol
+    if "://" in url:
+        old_protocol = url.split("://")[0]
+        if old_protocol != protocol:
+            return url.replace(f"{old_protocol}://", f"{protocol}://", 1)
+
+    return url
