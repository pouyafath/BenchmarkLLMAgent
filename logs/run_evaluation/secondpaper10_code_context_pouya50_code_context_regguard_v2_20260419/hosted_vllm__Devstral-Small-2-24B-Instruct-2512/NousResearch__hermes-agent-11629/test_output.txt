+def _strip_discord_mentions(content: str) -> str:
+    """Strip all Discord mention syntax from *content*.
+
+    Removes:
+    - User mentions: ``<@123>`` or ``<@!123>``
+    - Role mentions: ``<@&123>``
+    - Channel mentions: ``<#123>``
+    - Everyone/here mentions: ``@everyone`` and ``@here``
+
+    Returns the cleaned content with all mention syntax removed.
+    """
+    if not content:

+
+    # Remove user mentions: <@123> or <@!123>
+    content = re.sub(r'<@!?\d+>', '', content)
+    # Remove role mentions: <@&123>
+    content = re.sub(r'<@&\d+>', '', content)
+    # Remove channel mentions: <#123>
+    content = re.sub(r'<#\d+>', '', content)
+    # Remove @everyone and @here
+    content = re.sub(r'@(everyone|here)', '', content, flags=re.IGNORECASE)
+
+    return content.strip()
