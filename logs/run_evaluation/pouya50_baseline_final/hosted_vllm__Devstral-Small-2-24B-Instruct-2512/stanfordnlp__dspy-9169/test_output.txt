+# Wrapper class to adapt DSPy LM to GEPA's LanguageModel protocol
+class DSPyLanguageModelWrapper:
+    """Wrapper to adapt DSPy LM to GEPA's LanguageModel protocol.
+
+    GEPA expects a LanguageModel that returns str, but DSPy LM returns:
+    - list[str] for simple completions
+    - list[dict[str, Any]] for completions with reasoning/tool calls
+
+    This wrapper extracts the text content from the DSPy LM output.
+    """
+    def __init__(self, lm):
+        self.lm = lm
+
+    def __call__(self, prompt: str) -> str:
+        """Call the wrapped LM and extract text content.
+
+        Args:
+            prompt: The prompt to send to the LM
+
+        Returns:
+            str: The extracted text content from the LM output
+        """
+        lm_output = self.lm(prompt)
+        if isinstance(lm_output, list) and len(lm_output) > 0:
+            first_output = lm_output[0]
+            if isinstance(first_output, dict):
+                # Extract text from dict (DSPy 3.1.0+ format with reasoning)
+                return first_output.get("text", "")
+            elif isinstance(first_output, str):
+                # Extract text from string (legacy format)
+                return first_output
+        return ""
